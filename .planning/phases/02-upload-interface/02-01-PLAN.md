```html
<!DOCTYPE html>

    <!-- 1. Drop zone: a <div> containing a hidden <input type="file"> -->
    <div id="drop-zone" class="drop-zone" role="button" tabindex="0"
         aria-label="Drop PDF here or click to browse">
      <input id="file-input" type="file" accept=".pdf,application/pdf"
             aria-label="Select a PDF file" hidden>
      <p class="drop-zone__prompt">Drag & drop your PDF here</p>
      <button id="browse-btn" type="button">Browse file</button>
      <!-- Size hint visible before selection (US-0.4 requirement) -->
      <p class="drop-zone__hint">Maximum file size: 50 MB</p>
    </div>

    <!-- 2. Filename + size display area (hidden until file selected) -->
    <div id="file-info" class="file-info" hidden>
      <span id="file-name"></span>
      <span id="file-size"></span>
    </div>

    <!-- 3. Convert button (disabled until valid file selected) -->
    <button id="convert-btn" type="button" disabled>Convert to DOCX</button>

    <!-- 4. <progress> element for upload progress (hidden during IDLE) -->
    <progress id="upload-progress" value="0" max="100" hidden
              aria-label="Upload progress"></progress>

    <!-- 5. Status banner with aria-live="polite" for screen reader announcements -->
    <div id="status-banner" class="status-banner" aria-live="polite" aria-atomic="true"></div>

    <!-- 6. Error detail <p> (hidden unless in ERROR state) -->
    <p id="error-detail" class="error-detail" hidden></p>

    <!-- 7. Try Again button (visible in ERROR state) -->
    <button id="try-again-btn" type="button" hidden>Try Again</button>

    <!-- 8. Convert Another File anchor (visible in SUCCESS state) -->
    <a id="convert-another-link" href="#" hidden>Convert Another File</a>
  </main>

  <script src="app.js"></script>
</body>
</html>
```

**Key requirements:**
- `id` attributes are exactly as shown — app.js will reference them by these IDs
- Drop zone `<div>` has `role="button"` and `tabindex="0"` for keyboard accessibility (US-0.6)
- `<input type="file">` has `accept=".pdf,application/pdf"` (US-0.1)
- Size hint paragraph with "Maximum file size: 50 MB" is ALWAYS visible before selection (US-0.4)
- `<progress>` element (NOT a div — a real `<progress>` HTML element) for XHR progress events
- Status banner `<div>` has BOTH `aria-live="polite"` AND `aria-atomic="true"`
- `<script src="app.js">` at BOTTOM of `<body>`, AFTER the main content
- No inline styles in HTML — all styling via styles.css classes

```typescript
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  testDir: './e2e',
  timeout: 10000,
  expect: { timeout: 5000 },
  fullyParallel: false,
  retries: 0,
  reporter: 'list',
  use: {
    baseURL: 'http://localhost:8000',
    headless: true,
    screenshot: 'only-on-failure',
    trace: 'off',
  },
  projects: [
    {
      name: 'chromium',
      use: { ...devices['Desktop Chrome'] },
    },
  ],
  // Auto-start uvicorn before tests; reuse if already running
  webServer: {
    command: 'uvicorn app.main:app --host 0.0.0.0 --port 8000',
    url: 'http://localhost:8000/api/health',
    reuseExistingServer: true,
    timeout: 15000,
  },
});
```
