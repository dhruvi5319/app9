"""
app/core/cleanup.py
-------------------
CleanupService — periodic TTL sweep that deletes orphaned job directories.

Design:
  - sweep() is an async method but does synchronous filesystem I/O (acceptable:
    sweep runs infrequently, at most once every SWEEP_INTERVAL_MINUTES minutes,
    and the I/O is bounded by the number of active jobs).
  - Age is determined by directory mtime (st_mtime), which is set when the
    directory is created by ConversionService and does not change thereafter.
  - Both the temp directory subtree AND the registry entry are removed atomically
    from the sweep's perspective (registry.delete is idempotent — returns False
    if already gone, which is fine).
  - sweep() returns a dict with 'deleted' and 'freed_bytes' for observability.
  - No file content is ever logged — only counts and byte totals.
"""

from __future__ import annotations

import logging
import shutil
import time
from pathlib import Path
from typing import TypedDict

from app.core.config import settings
from app.core.registry import job_registry

logger = logging.getLogger(__name__)


class SweepResult(TypedDict):
    deleted: int
    freed_bytes: int


class CleanupService:
    """Scans TEMP_DIR and removes job directories older than TTL_MINUTES.

    Usage::

        svc = CleanupService()
        result = await svc.sweep()
        # result == {"deleted": 3, "freed_bytes": 2097152}
    """

    async def sweep(self) -> SweepResult:
        """Scan TEMP_DIR and delete subdirectories that have exceeded TTL.

        For each deleted directory:
          - Measures total bytes freed (sum of all file sizes).
          - Removes the corresponding JobRegistry entry (no-op if already gone).

        Returns:
            SweepResult with 'deleted' count and 'freed_bytes' total.
        """
        temp_dir = Path(settings.TEMP_DIR)
        now = time.time()
        ttl_seconds = settings.TTL_MINUTES * 60
        deleted_count = 0
        freed_bytes = 0

        if not temp_dir.is_dir():
            logger.warning(
                "TTL sweep: TEMP_DIR '%s' does not exist — skipping sweep",
                temp_dir,
            )
            return {"deleted": 0, "freed_bytes": 0}

        try:
            entries = list(temp_dir.iterdir())
        except PermissionError as exc:
            logger.error("TTL sweep: cannot read TEMP_DIR '%s': %s", temp_dir, exc)
            return {"deleted": 0, "freed_bytes": 0}

        for entry in entries:
            if not entry.is_dir():
                # Skip any stray files at the top level of TEMP_DIR
                continue

            try:
                # Use directory mtime as the creation proxy.
                # ConversionService creates the directory once and never modifies it,
                # so mtime ≈ creation time on all POSIX systems.
                age_seconds = now - entry.stat().st_mtime
            except OSError as exc:
                logger.warning(
                    "TTL sweep: cannot stat directory '%s': %s — skipping",
                    entry,
                    exc,
                )
                continue

            if age_seconds <= ttl_seconds:
                continue  # Still within TTL — leave it

            # ── Measure bytes before deletion ─────────────────────────────
            dir_bytes = 0
            try:
                dir_bytes = sum(
                    f.stat().st_size
                    for f in entry.rglob("*")
                    if f.is_file()
                )
            except OSError:
                # Non-fatal — size accounting may be incomplete
                pass

            # ── Delete directory tree ──────────────────────────────────────
            try:
                shutil.rmtree(entry, ignore_errors=False)
                logger.info(
                    "TTL sweep: deleted '%s' (age=%.0fs, size=%d bytes)",
                    entry.name,
                    age_seconds,
                    dir_bytes,
                )
            except Exception as exc:
                logger.error(
                    "TTL sweep: failed to delete '%s': %s — skipping",
                    entry.name,
                    exc,
                )
                continue  # Don't increment counters if deletion failed

            # ── Remove registry entry (idempotent) ────────────────────────
            removed = job_registry.delete(entry.name)
            if removed:
                logger.debug(
                    "TTL sweep: removed registry entry for '%s'", entry.name
                )

            deleted_count += 1
            freed_bytes += dir_bytes

        logger.info(
            "TTL sweep complete: deleted=%d dirs, freed=%d bytes",
            deleted_count,
            freed_bytes,
        )
        return {"deleted": deleted_count, "freed_bytes": freed_bytes}


# Module-level singleton — imported by main.py lifespan
cleanup_service = CleanupService()
```

Key implementation notes:
- `entry.stat().st_mtime` is used as the age proxy (not `st_ctime`) — consistent on Linux.
- `shutil.rmtree(entry, ignore_errors=False)` — we want to know if deletion fails; failures are caught and logged but do NOT crash the sweep loop.
- `job_registry.delete(entry.name)` uses the directory name as the job_id (UUIDs are both the dir name and the job_id).
- Stray files (non-directories) at the TEMP_DIR top level are silently skipped.
- No file content is logged — only names, ages, and byte counts.

async def _run_sweep_loop(svc) -> None:
    """Periodic TTL sweep loop — runs forever until cancelled.

    Sleeps first (SWEEP_INTERVAL_MINUTES), then sweeps. This means the first
    sweep happens after the initial interval, not immediately on startup —
    intentional, as files just written at startup are never immediately expired.
    """
    while True:
        await asyncio.sleep(settings.SWEEP_INTERVAL_MINUTES * 60)
        try:
            result = await svc.sweep()
            logger.debug("Sweep result: %s", result)
        except Exception as exc:
            logger.error("TTL sweep error: %s", exc)


@asynccontextmanager
async def lifespan(app: FastAPI) -> AsyncIterator[None]:
    # ── Startup ───────────────────────────────────────────────────────────
    os.makedirs(settings.TEMP_DIR, exist_ok=True)
    logger.info("Server started. TEMP_DIR=%s", settings.TEMP_DIR)

    # Start TTL sweep background task
    from app.core.cleanup import cleanup_service
    sweep_task = asyncio.create_task(_run_sweep_loop(cleanup_service))
    logger.info(
        "TTL sweep task started (interval=%dm, ttl=%dm)",
        settings.SWEEP_INTERVAL_MINUTES,
        settings.TTL_MINUTES,
    )

    yield

    # ── Shutdown ──────────────────────────────────────────────────────────
    sweep_task.cancel()
    try:
        await sweep_task
    except asyncio.CancelledError:
        pass  # Expected — task was cancelled cleanly
    logger.info("TTL sweep task stopped.")
    logger.info("Server shutting down.")
```
