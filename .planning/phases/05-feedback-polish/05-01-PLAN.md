**Change 10 — Add download-btn DOM reference and click handler:**

After the Convert Another File handler block (end of file, before `// ─── Initial State`), add:

```javascript
// ─── Download DOCX Button Handler (SUCCESS state re-download) ─────────────────
// The download-btn is shown in SUCCESS state. Clicking it re-triggers the download
// via the stored lastJobId. If the file was already deleted (Phase 4 one-time
// download policy), the server returns 404 and the browser follows the redirect
// to an error. We detect this by listening for the fetch response before using
// the invisible anchor, so we can transition SUCCESS→ERROR with JOB_NOT_FOUND.
if (downloadBtn) {
  downloadBtn.addEventListener('click', () => {
    if (!lastJobId) return;
    // Use fetch to probe the download endpoint first (HEAD would be ideal but
    // the server returns 200 with body on GET). We use fetch with redirect:'manual'
    // so a 404 is caught before the browser opens a download dialog.
    fetch('/api/download/' + lastJobId, { method: 'GET', redirect: 'manual' })
      .then((response) => {
        if (response.ok) {
          // File still available — trigger download via anchor
          triggerDownload(lastJobId, lastFilename);
        } else if (response.status === 404) {
          // File was already deleted after first download
          setState('ERROR', ERROR_MESSAGES.JOB_NOT_FOUND);
          if (ERROR_DETAILS.JOB_NOT_FOUND) {
            errorDetail.textContent = ERROR_DETAILS.JOB_NOT_FOUND;
            errorDetail.hidden = false;
          }
        } else {
          setState('ERROR', ERROR_MESSAGES.INTERNAL_ERROR);
        }
      })
      .catch(() => {
        setState('ERROR', 'Upload failed. Please check your connection and try again.');
      });
  });
}
```
