// Helper: create a fake PDF file
function createFakePdf(name: string, sizeBytes = 1024): string {
  const header = Buffer.from('%PDF-1.4\n', 'utf8');
  const padding = Buffer.alloc(Math.max(0, sizeBytes - header.length), 0x20);
  const tmpPath = path.join('/tmp', name);
  fs.writeFileSync(tmpPath, Buffer.concat([header, padding]));
  return tmpPath;
